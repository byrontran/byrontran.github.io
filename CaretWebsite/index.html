Have a Taste:

fibonacci.caret 
def fibonacci {
  lambda n :
    (if (= n 0)
      0
        (if (= n 1)
          1
          (+ (fibonacci (- n 1)) (fibonacci (- n 2)))))
};
  
def fibonacci-sequence {
  lambda n :
  (if (= n 0)
    (list)
    (if (= n 1)
      (list 0)
      (construct (fibonacci (- n 1)) (fibonacci-sequence (- n 1)))))
};
  
print-result(fibonacci-sequence 5);
